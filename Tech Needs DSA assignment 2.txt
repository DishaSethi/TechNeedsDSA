      for (int j = 1; j <= sp; j++)  {
        System.out.print("  ");
      }
      int cval = val;
      for (int j = 1; j <= st; j++)  {
        System.out.print(cval + " "); //cval++;
        if ( j <= st / 2)
        {
          cval++;
        }
        else
        {
          cval--;
        }
      }


      if (i <= n / 2)

        st += 2;
        val++;

        sp++;

      }
